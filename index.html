async function publishData(){ const token=(document.getElementById('githubToken').value.trim()||localStorage.getItem('gh_token')||'').trim(); if(!token){toast('GitHub token required to publish','error');return;} setLoading(true,'Merging and saving data…'); try{ let merged=JSON.parse(JSON.stringify(STATE.agents)),roster={...STATE.roster},allDates=new Set(STATE.allDates); if(pendingRoster)Object.assign(roster,pendingRoster); if(pendingWisp){ pendingWisp.dateCols.forEach(d=>allDates.add(d)); const map={};merged.forEach(a=>{map[a.email]=a;}); pendingWisp.agents.forEach(na=>{ if(map[na.email]){ Object.assign(map[na.email].daily,na.daily); map[na.email].name=na.name; } else { map[na.email]={...na}; merged.push(map[na.email]); } }); } // ── ROSTER IS SOURCE OF TRUTH ── // Remove any agent not present in the current roster. This runs on every // publish so agents removed from the roster are pruned from history too. const rosterEmails=new Set(Object.keys(roster)); const beforeCount=merged.length; merged=merged.filter(a=>rosterEmails.has(a.email)); const removed=beforeCount-merged.length; if(removed>0)toast(`Removed ${removed} agent${removed>1?'s':''} not on roster`,'info'); merged.forEach(a=>{a.team=roster[a.email]||'un';}); const sortedDates=[...allDates].sort(); const payload={agents:merged,roster,allDates:sortedDates,lastUpdated:new Date().toISOString()}; let sha=null; try{const sr=await fetch(`https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${DATA_PATH}`,{headers:{Authorization:`token ${token}`,Accept:'application/vnd.github.v3+json'}});if(sr.ok){const j=await sr.json();sha=j.sha;}}catch(e){} const body={message:`Update dashboard data — ${new Date().toLocaleDateString()}`,content:btoa(unescape(encodeURIComponent(JSON.stringify(payload))))}; if(sha)body.sha=sha; const res=await fetch(`https://api.github.com/repos/${REPO_OWNER}/${REPO_NAME}/contents/${DATA_PATH}`,{method:'PUT',headers:{Authorization:`token ${token}`,Accept:'application/vnd.github.v3+json','Content-Type':'application/json'},body:JSON.stringify(body)}); if(!res.ok){const err=await res.json();throw new Error(err.message||'GitHub API error');} localStorage.setItem('gh_token',token); applyData(payload);pendingWisp=null;pendingRoster=null; closeDrawer();toast('Data saved and published!','success'); }catch(e){toast('Error: '+e.message,'error');} setLoading(false); }
